### Add to either:
- `~/.config/nvim/init.lua` (single file)
- `~/.config/nvim/lua/keybindings.lua` (modular)

### Basic Operations
- Delete all in file without overwriting clipboard
- Select all
- Copy selection
- Paste clipboard
- Start selecting to the Left
- Start selecting to the right

```lua
-- Copy selection (Ctrl+c)
vim.keymap.set('v', '<C-c>', '"+y', { noremap = true, silent = true })

-- Paste from clipboard (Ctrl+v)
vim.keymap.set('n', '<C-v>', '"+p', { noremap = true, silent = true })
vim.keymap.set('i', '<C-v>', '<C-r>+', { noremap = true, silent = true })

-- Select all (Ctrl+a)
vim.keymap.set('n', '<C-a>', 'ggVG', { noremap = true, silent = true })
```

### Undo/Redo

```lua
-- Undo (Ctrl+z)
vim.keymap.set('n', '<C-z>', 'u', { noremap = true, silent = true })
vim.keymap.set('i', '<C-z>', '<C-o>u', { noremap = true, silent = true })

-- Redo (Ctrl+Shift+z)
vim.keymap.set('n', '<C-S-z>', '<C-r>', { noremap = true, silent = true })
vim.keymap.set('i', '<C-S-z>', '<C-o><C-r>', { noremap = true, silent = true })
```

### Text Selection

```lua
-- Shift + Arrow keys for selection
vim.keymap.set('n', '<S-Left>', 'v<Left>', { noremap = true, silent = true })
vim.keymap.set('n', '<S-Right>', 'v<Right>', { noremap = true, silent = true })
vim.keymap.set('n', '<S-Up>', 'vk', { noremap = true, silent = true })
vim.keymap.set('n', '<S-Down>', 'vj', { noremap = true, silent = true })

-- Insert mode selections
vim.keymap.set('i', '<S-Left>', '<Esc>v<Left>', { noremap = true, silent = true })
vim.keymap.set('i', '<S-Right>', '<Esc>v<Right>', { noremap = true, silent = true })
vim.keymap.set('i', '<S-Up>', '<Esc>vk', { noremap = true, silent = true })
vim.keymap.set('i', '<S-Down>', '<Esc>vj', { noremap = true, silent = true })
```

For modular setup, create keybindings.lua and add to init.lua:
```lua
require('keybindings').setup()
```

If Ctrl+Shift+Z doesn't work, replace with:
```lua
vim.keymap.set('n', '<C-y>', '<C-r>', { noremap = true, silent = true })
vim.keymap.set('i', '<C-y>', '<C-o><C-r>', { noremap = true, silent = true })
```
